        function register(event) {
            event.preventDefault();
            
            if (!selectedRole) {
                alert('Please select a role');
                return;
            }

            const name = document.getElementById('registerName').value;
            const email = document.getElementById('registerEmail').value;
            const password = document.getElementById('registerPassword').value;
            const employeeId = document.getElementById('registerEmployeeId').value;

            // Check if email already exists
            if (DB.users.find(u => u.email === email)) {
                alert('Email already registered');
                return;
            }

            // For employee registration, verify employee ID
            if (selectedRole === 'employee') {
                const employeeExists = DB.employees.find(e => e.id == employeeId && e.email === email);
                if (!employeeExists) {
                    alert('Invalid Employee ID or email mismatch');
                    return;
                }
            }

            const newUser = {
                id: Date.now(),
                name,
                email,
                password, // In production, hash this!
                role: selectedRole,
                employeeId: selectedRole === 'employee' ? employeeId : null
            };

            DB.users.push(newUser);
            DB.save('users');

            alert('Registration successful! Please login.');
            switchAuthTab('login');
            event.target.reset();
        }

        function calculateLeaveBalance(employeeId) {
            const approvedLeave = DB.leaveRequests.filter(l => l.employeeId == employeeId && l.status === 'approved');
            
            let annual = 15;
            let sick = 10;
            let personal = 5;

            approvedLeave.forEach(leave => {
                const days = Math.ceil((new Date(leave.endDate) - new Date(leave.startDate)) / (1000 * 60 * 60 * 24)) + 1;
                if (leave.type === 'annual') annual -= days;
                else if (leave.type === 'sick') sick -= days;
                else if (leave.type === 'personal') personal -= days;
            });

            return { annual: Math.max(0, annual), sick: Math.max(0, sick), personal: Math.max(0, personal) };
        }

        function checkIn() {
            const today = new Date().toISOString().split('T')[0];
            const existing = DB.attendance.find(a => a.employeeId == currentUser.employeeId && a.date === today);
            
            if (existing) {
                alert('You have already checked in today!');
                return;
            }

            const now = new Date();
            const time = now.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit', hour12: true });

            DB.attendance.push({
                id: Date.now(),
                employeeId: currentUser.employeeId,
                date: today,
                checkIn: time,
                checkOut: null
            });
            DB.save('attendance');

            alert('Checked in successfully at ' + time);
            loadMyAttendance();
        }

        function checkOut() {
            const today = new Date().toISOString().split('T')[0];
            const existing = DB.attendance.find(a => a.employeeId == currentUser.employeeId && a.date === today);
            
            if (!existing) {
                alert('Please check in first!');
                return;
            }

            if (existing.checkOut) {
                alert('You have already checked out today!');
                return;
            }

            const now = new Date();
            const time = now.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit', hour12: true });

            existing.checkOut = time;
            DB.save('attendance');

            alert('Checked out successfully at ' + time);
            loadMyAttendance();
        }

        function requestLeave(leaveType) {
            const balance = calculateLeaveBalance(currentUser.employeeId);
            
            if (balance[leaveType] <= 0) {
                alert(`You have no ${leaveType} leave days remaining!`);
                return;
            }

            const startDate = prompt(`Enter start date (YYYY-MM-DD):`);
            if (!startDate) return;

            const endDate = prompt(`Enter end date (YYYY-MM-DD):`);
            if (!endDate) return;

            const reason = prompt(`Enter reason for leave:`);
            if (!reason) return;

            DB.leaveRequests.push({
                id: Date.now(),
                employeeId: currentUser.employeeId,
                type: leaveType,
                startDate,
                endDate,
                reason,
                status: 'pending',
                requestDate: new Date().toISOString().split('T')[0]
            });
            DB.save('leaveRequests');

            alert('Leave request submitted successfully!');
            loadMyLeave();
        }

        function approveProfileUpdate(requestId) {
            const request = DB.profileRequests.find(r => r.id === requestId);
            if (!request) return;

            const emp = DB.employees.find(e => e.id == request.employeeId);
            if (emp) {
                Object.assign(emp, request.updates);
                request.status = 'approved';
                DB.save('employees');
                DB.save('profileRequests');
                alert('Profile update approved!');
                loadProfileRequests();
            }
        }

        function rejectProfileUpdate(requestId) {

                request.status = 'rejected';
                DB.save('profileRequests');
                alert('Profile update rejected');
                loadProfileRequests();
            }
        }

            if (email) updates.email = email;

                return;
