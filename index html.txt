    <script>
        // Global state
        let currentUser = null;
        let selectedRole = null;
        
        // Data storage
        const DB = {
            users: JSON.parse(localStorage.getItem('hrm_users') || '[]'),
            employees: JSON.parse(localStorage.getItem('hrm_employees') || '[]'),
            attendance: JSON.parse(localStorage.getItem('hrm_attendance') || '[]'),
            leaveRequests: JSON.parse(localStorage.getItem('hrm_leave') || '[]'),
            
            save(key) {
                localStorage.setItem('hrm_' + key, JSON.stringify(this[key]));
            }
        };

        // Initialize demo data
        if (DB.employees.length === 0) {
            DB.employees = [
                {id: 1, name: 'Emma Wilson', email: 'emma@company.com', position: 'Designer', department: 'Design', status: 'active', phone: '123456'},
                {id: 2, name: 'Michael Chen', email: 'michael@company.com', position: 'Developer', department: 'Engineering', status: 'active', phone: '123457'},
                {id: 3, name: 'Sarah Davis', email: 'sarah@company.com', position: 'Manager', department: 'Product', status: 'active', phone: '123458'}
            ];
            DB.save('employees');
        }

        // Auth tab switching
        document.getElementById('loginTab').addEventListener('click', function() {
            document.getElementById('loginTab').classList.add('active');
            document.getElementById('registerTab').classList.remove('active');
            document.getElementById('loginForm').classList.remove('hidden');
            document.getElementById('registerForm').classList.add('hidden');
        });

        document.getElementById('registerTab').addEventListener('click', function() {
            document.getElementById('registerTab').classList.add('active');
            document.getElementById('loginTab').classList.remove('active');
            document.getElementById('registerForm').classList.remove('hidden');
            document.getElementById('loginForm').classList.add('hidden');
        });

        // Role selection for login
        document.getElementById('loginEmployer').addEventListener('click', function() {
            selectedRole = 'employer';
            document.querySelectorAll('#loginForm .role-option').forEach(el => el.classList.remove('selected'));
            this.classList.add('selected');
        });

        document.getElementById('loginEmployee').addEventListener('click', function() {
            selectedRole = 'employee';
            document.querySelectorAll('#loginForm .role-option').forEach(el => el.classList.remove('selected'));
            this.classList.add('selected');
        });

        // Role selection for register
        document.getElementById('registerEmployer').addEventListener('click', function() {
            selectedRole = 'employer';
            document.querySelectorAll('#registerForm .role-option').forEach(el => el.classList.remove('selected'));
            this.classList.add('selected');
            document.getElementById('employeeIdGroup').classList.add('hidden');
        });

        document.getElementById('registerEmployee').addEventListener('click', function() {
            selectedRole = 'employee';
            document.querySelectorAll('#registerForm .role-option').forEach(el => el.classList.remove('selected'));
            this.classList.add('selected');
            document.getElementById('employeeIdGroup').classList.remove('hidden');
        });

        // Login
        document.getElementById('loginFormElement').addEventListener('submit', function(e) {
            e.preventDefault();
            
            if (!selectedRole) {
                alert('Please select a role');
                return;
            }

            const email = document.getElementById('loginEmail').value;
            const password = document.getElementById('loginPassword').value;

            const user = DB.users.find(u => u.email === email && u.password === password && u.role === selectedRole);

            if (!user) {
                alert('Invalid credentials');
                return;
            }

            currentUser = user;
            showMainApp();
        });

        // Register
        document.getElementById('registerFormElement').addEventListener('submit', function(e) {
            e.preventDefault();
            
            if (!selectedRole) {
                alert('Please select a role');
                return;
            }

            const name = document.getElementById('registerName').value;
            const email = document.getElementById('registerEmail').value;
            const password = document.getElementById('registerPassword').value;
            const employeeId = document.getElementById('registerEmployeeId').value;

            if (DB.users.find(u => u.email === email)) {
                alert('Email already registered');
                return;
            }

            if (selectedRole === 'employee' && !employeeId) {
                alert('Please enter Employee ID');
                return;
            }

            const newUser = {
                id: Date.now(),
                name,
                email,
                password,
                role: selectedRole,
                employeeId: selectedRole === 'employee' ? employeeId : null
            };

            DB.users.push(newUser);
            DB.save('users');

            alert('Registration successful! Please login.');
            document.getElementById('loginTab').click();
            this.reset();
        });

        // Logout
        document.getElementById('logoutBtn').addEventListener('click', function() {
            currentUser = null;
            selectedRole = null;
            document.getElementById('mainApp').classList.add('hidden');
            document.getElementById('authScreen').classList.remove('hidden');
        });

        // Show main app

            loadDashboard();
        }

                        <div class="stat-card" onclick="loadMyAttendance()">

                `;
            }
        }

        // Load Employees
        function loadEmployees() {
            if (currentUser.role !== 'employer') return;
            updateActiveNav(1);
            
            const content = document.getElementById('contentArea');
            const empList = DB.employees.map(emp => `
                <div class="employee-list-item" onclick="viewEmployee(${emp.id})">
                    <div class="employee-avatar-small">${emp.name.charAt(0)}</div>
                    <div style="flex: 1;">
                        <div style="font-weight: 600;">${emp.name}</div>
                        <div style="font-size: 0.85rem; color: #64748B;">${emp.position}</div>
                    </div>
                    <span class="badge success">${emp.status}</span>
                </div>
            `).join('');

            content.innerHTML = `
                <h1 class="page-title">Employee Management</h1>
                <div class="card">
                    <div class="card-header">
                        <h2 class="card-title">All Employees (${DB.employees.length})</h2>

        // Check In
        function checkIn() {
            const today = new Date().toISOString().split('T')[0];
            const existing = DB.attendance.find(a => a.employeeId == currentUser.employeeId && a.date === today);
            
            if (existing) {
                alert('Already checked in!');
                return;
            }

            const time = new Date().toLocaleTimeString('en-US', {hour: '2-digit', minute: '2-digit'});
            DB.attendance.push({
                id: Date.now(),
                employeeId: currentUser.employeeId,
                date: today,
                checkIn: time,
                checkOut: null
            });
            DB.save('attendance');
            alert('Checked in at ' + time);
            loadMyAttendance();
        }

        // Check Out
        function checkOut() {
            const today = new Date().toISOString().split('T')[0];
            const existing = DB.attendance.find(a => a.employeeId == currentUser.employeeId && a.date === today);
            
            if (!existing) {
                alert('Please check in first!');
                return;
            }

            if (existing.checkOut) {
                alert('Already checked out!');
                return;
            }

            const time = new Date().toLocaleTimeString('en-US', {hour: '2-digit', minute: '2-digit'});
            existing.checkOut = time;
            DB.save('attendance');
            alert('Checked out at ' + time);
            loadMyAttendance();
        }
